7   import spark.ModelAndView;
8   import spark.QueryParamsMap;
9   import spark.Request;
10  import spark.Response;
11  
12  import com.google.common.collect.ImmutableMap;
13  
14  import edu.brown.cs.dpark20.bacon.Actor;
15  import edu.brown.cs.dpark20.bacon.BaconREPL;
16  import edu.brown.cs.dpark20.bacon.Movie;
17  import edu.brown.cs.dpark20.bacon.Path;
18  import edu.brown.cs.dpark20.graph.GraphAlgorithms;
19  import edu.brown.cs.dpark20.repl.REPL;
20  
21  /**
22   * Handler for bacon's gui page.
23   *
24   */
25  public class BaconHandler {
26  
27    private static REPL repl;
28  
29    /**
30     * constructor for handler.
31     *
32     * @param r
33     *          repl to use.
34     */
35    public BaconHandler(REPL r) {
36      repl = r;
37    }
38  
39    /**
40     * handle for the page.
41     *
42     * @param req
43     *          request
44     * @param res
45     *          response
46     * @return ModelAndView for page.
47     */
48    public ModelAndView handle(Request req, Response res) {
49      BaconREPL baconrepl = repl.getBaconRepl();
50      String givenCmd;
51      QueryParamsMap qm = req.queryMap();
52  
53      String startActor = qm.value("startActor");
54      String endActor = qm.value("endActor");
55  
56      // create and evaluate query in this method!
57      givenCmd = "connect \"" + startActor + "\" \"" + endActor + "\"";
58  
59      List<String> toReturn = new ArrayList<>();
60      List<String> actIDs = new ArrayList<>();
61      List<String> movNames = new ArrayList<>();
62      List<String> movIDs = new ArrayList<>();
63  
64      ImmutableMap.Builder<String, Object> builder;
65      builder = new ImmutableMap.Builder<>();
66      builder.put("title", "Bacon: Query");
67      Map<String, Object> variables;
68      GraphAlgorithms query = baconrepl.getQuery();
69      if (query == null) {
70        toReturn.add("Load sql file in REPL first!");
71        builder.put("results", toReturn);
72        builder.put("result", toReturn.get(0));
73        variables = builder.build();
74        return new ModelAndView(variables, "baconpath.ftl");
75      }
76  
77      baconrepl.parseAndEvaluate(givenCmd, true, true);
78      Path<Actor, Movie> resultsFound = baconrepl.getResults();
79      if (resultsFound == null || resultsFound.getVertices().size() == 0) {
80        // if null or actor compared with itself
81        toReturn.add("Query didn't find a path! (REPL might have more details)");
82        builder.put("results", toReturn);
83        builder.put("result", toReturn.get(0));
84        variables = builder.build();
85        return new ModelAndView(variables, "baconpath.ftl");
86      } else if (resultsFound.getVertices().size() != 0
87          && resultsFound.getEdges().size() == 0) {
88        // if no path found
89        StringBuilder s = new StringBuilder();
90        s.append(startActor);
91        s.append(" doesn't link to ");
92        s.append(endActor);
93  
94        String f = "<a href=\"/bacon/actor/:id"
95            + resultsFound.getVertices().get(0) + "\">" + startActor + "</a>"
96            + "-/-<a href=\"/bacon/actor/:id" + resultsFound.getVertices().get(1)
97            + "\">" + endActor + "</a>";
98  
99        toReturn.add(f);
100       builder.put("results", toReturn);
101       builder.put("result", toReturn.get(0));
102       variables = builder.build();
103       return new ModelAndView(variables, "baconpath.ftl");
104     } else {
105       // otherwise, path was found
106       int count = 0;
107       List<Actor> acts = resultsFound.getVertices();
108       for (Movie mov : resultsFound.getEdges()) {
109         StringBuilder s = new StringBuilder();
110         s.append(acts.get(count).getName());
111         toReturn.add(acts.get(count).getName());
112         actIDs.add(acts.get(count).getId().replace("/", "$"));
113         s.append(" -> ");
114         s.append(acts.get(count + 1).getName());
115         toReturn.add(acts.get(count + 1).getName());
116         actIDs.add(acts.get(count + 1).getId().replace("/", "$"));
117         s.append(" : ");
118         s.append(mov.getName());
119         movNames.add(mov.getName());
120         movIDs.add(mov.getId().replace("/", "$"));
121         // toReturn.add(s.toString());
122         count++;
123       }
124     }
125 
126     // builder.put("actnames", toReturn);
127     // builder.put("actids", actIDs);
128     // builder.put("movnames", movNames);
129     // builder.put("movIDs", movIDs);
130     if (movIDs.size() == 0) {
131       System.out.println("here");
132     }
133 
134     List<String> send = new ArrayList<>();
135     for (int i = 0; i < movIDs.size(); i += 1) {
136 
137       String f = "<a href=\"/bacon/actor/:id" + actIDs.get(2 * i) + "\">"
138           + toReturn.get(2 * i) + "</a>" + "-> <a href=\"/bacon/actor/:id"
139           + actIDs.get(2 * i + 1) + "\">" + toReturn.get(2 * i + 1) + "</a>"
140           + ": <a href=\"/bacon/movie/:id" + movIDs.get(i) + "\">"
141           + movNames.get(i) + "</a>";
142       send.add(f);
143 
144     }
145 
146     builder.put("results", send);
147     builder.put("result", send.get(0));
148 
149     variables = builder.build();
150 
151     return new ModelAndView(variables, "baconpath.ftl");
152   }
153 }
