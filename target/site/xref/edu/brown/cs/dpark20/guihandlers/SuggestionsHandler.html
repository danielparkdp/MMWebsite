3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.Map;
6   
7   import spark.QueryParamsMap;
8   import spark.Request;
9   import spark.Response;
10  
11  import com.google.common.collect.ImmutableMap;
12  import com.google.gson.Gson;
13  
14  import edu.brown.cs.dpark20.autocorrect.AutocorrectREPL;
15  
16  /**
17   * SuggestionsHandler for processing the gui suggestions.
18   *
19   */
20  public class SuggestionsHandler {
21  
22    private static final Gson GSON = new Gson();
23  
24    private static AutocorrectREPL acrepl;
25  
26    /**
27     * Constructor for handler.
28     *
29     * @param r
30     *          repl to put.
31     */
32    public SuggestionsHandler(AutocorrectREPL r) {
33      acrepl = r;
34  
35    }
36  
37    /**
38     * handles the request and sends response.
39     *
40     * @param req
41     *          request
42     * @param res
43     *          response
44     * @return String to manage values given.
45     */
46    public String handle(Request req, Response res) {
47  
48      QueryParamsMap qm = req.queryMap();
49      String words = qm.value("word");
50  
51      // check if empty or whitespace at end
52      if (words.trim().equals("")
53          || Character.isWhitespace(words.toCharArray()[words.length() - 1])) {
54        Map<String, Object> variables = ImmutableMap.of("map",
55            new ArrayList<String>());
56        return GSON.toJson(variables);
57      }
58  
59      acrepl.parseAndEvaluate("ac ".concat(words), "ac ".concat(words), true);
60      List<String> suggs = acrepl.getSuggestions();
61  
62      Map<String, Object> variables = ImmutableMap.of("map", suggs);
63      return GSON.toJson(variables);
64  
65    }
66  }
