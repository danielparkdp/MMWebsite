8   import spark.ModelAndView;

11  import spark.Response;

13  import com.google.common.collect.ImmutableMap;
14  
15  import edu.brown.cs.dpark20.dimensionpoint.DimensionPoint;
16  import edu.brown.cs.dpark20.kdtree.KDTree;
17  import edu.brown.cs.dpark20.repl.REPL;
18  import edu.brown.cs.dpark20.stars.Star;
19  
20  /**
21   * SubRHandler to handle submissions to Radius results.
22   *
23   */
24  public class SubRHandler {
25  
26    private static REPL repl;
27  
28    /**
29     * constructor, take repl to put in instance.
30     *
31     * @param r
32     *          repl to use.
33     */
34    public SubRHandler(REPL r) {
35      repl = r;
36    }
37  
38    /**
39     * handle for submissions of radius requests.
40     *
41     * @param req
42     *          request
43     * @param res
44     *          response
45     * @return ModelAndView for gui.
46     */
47    public ModelAndView handle(Request req, Response res) {
48      KDTree starsTree;
49      String givenCmd;
50      QueryParamsMap qm = req.queryMap();
51  
52      String num = qm.value("givenRadius");
53      String uniqueIden = qm.value("id");
54  
55      givenCmd = "radius " + num + " " + uniqueIden;
56  
57      List<String> foundStars = new ArrayList<String>();
58  
59      ImmutableMap.Builder<String, Object> builder;
60      builder = new ImmutableMap.Builder<String, Object>();
61      builder.put("title", "Stars: Query");
62      Map<String, Object> variables;
63      starsTree = repl.getStarsTree();
64      if (starsTree == null) {
65        foundStars.add("Get a tree created in REPL first!");
66        builder.put("results", foundStars);
67        builder.put("result", foundStars.get(0));
68        variables = builder.build();
69        return new ModelAndView(variables, "resultsr.ftl");
70      }
71  
72      repl.starsParseEval(givenCmd);
73      List<DimensionPoint> resultsFound = repl.getResult();
74      if (resultsFound == null) {
75        foundStars.add("Try again; input invalid (watch out for whitespace)");
76        builder.put("results", foundStars);
77        builder.put("result", foundStars.get(0));
78        variables = builder.build();
79        return new ModelAndView(variables, "resultsr.ftl");
80      }
81  
82      HashMap<String, String[]> starMap = repl.getStarsMap();
83      for (DimensionPoint neighbor : resultsFound) {
84        String[] info = starMap.get(((Star) neighbor).getName());
85        foundStars.add("Name of star is " + info[1] + ", with ID of " + info[0]
86            + ". Coords: " + info[2] + " " + info[3] + " " + info[4]);
87      }
88  
89      if (foundStars.size() == 0) {
90        foundStars.add("No stars matched");
91      }
92      builder.put("results", foundStars);
93      builder.put("result", foundStars.get(0));
94      variables = builder.build();
95  
96      return new ModelAndView(variables, "resultsr.ftl");
97  
98    }
99  
100 }
