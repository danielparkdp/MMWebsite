59    /**
60     * query the graph with dijkstra's.
61     *
62     * @param start
63     *          start vert.
64     * @param end
65     *          end vert.
66     * @return list of path.
67     */
68    @Override
69    public List<GraphVertex> graphQuery(GraphVertex start, GraphVertex end) {
70      // assert
71      assert (start != null);
72      assert (end != null);
73  
74      // If not have conn, just do pure. Otherwise do update.
75      if (conn1 == null) {
76        return this.pureDijkstra(start, end);
77      } else if (currVertices1.keySet().contains(start.getId())) {
78        // it is already found in cache
79        return this.updateDijkstra(currVertices1.get(start.getId()), end);
80      } else {
81        // not in cache, so query database
82        return this.updateDijkstra(start, end);
83      }
84    }

86    /**
87     * Method for when we simply want to use dijkstra's without adding to graph.
88     *
89     * @param start
90     *          starting vertex
91     * @param end
92     *          ending vertex
93     * @return List with vertex path
94     */
95    private List<GraphVertex> pureDijkstra(GraphVertex start, GraphVertex end) {
96      // dijkstra's algorithm with a pre-made graph
97      PriorityQueue<GraphVertex> pq = new PriorityQueue<GraphVertex>();
98      start.setDistance(0);
99      pq.add(start);
100     GraphVertex currVertex;
101     GraphVertex nextVertex;
102     boolean found = false;
103 
104     // go through priority queue.
105     while ((currVertex = pq.poll()) != null) {
106 
107       // add to currVertices1
108       currVertices1.put(currVertex.getId(), currVertex);
109 
110       if (currVertex.getId().equals(end.getId())) {
111         found = true;
112         end = currVertex;
113         break;
114       }
115 
116       // for each vertex, loop through edges
117       for (GraphEdge e : currVertex.getEdges()) {
118         nextVertex = e.getTo();
119 
120         if (nextVertex.getVisited()) {
121           continue;
122         }
123 
124         // determine if a better path was found for a vertex.
125         if ((currVertex.getDistance() + e.getWeight()) < nextVertex
126             .getDistance()) {
127           // reset this vertex
128           pq.remove(nextVertex);
129           nextVertex.setDistance((currVertex.getDistance() + e.getWeight()));
130           nextVertex.setPrev(currVertex);
131           pq.add(nextVertex);
132         }
133       }
134       // set as visited
135       currVertex.setVisited(true);
136     }
137 
138     // now, loop through from end and keep track of which vertex comes before.
139     List<GraphVertex> path = new ArrayList<GraphVertex>();
140     GraphVertex v = end;
141 
142     // if end not found, return empty
143     if (!found) {
144       // reset decorations at end.
145       for (GraphVertex vert : currVertices1.values()) {
146         vert.reset();
147       }
148       return path;
149     }
150 
151     while (v != null) {
152       // update path
153       path.add(v);
154       v = v.getPrev();
155     }
156     // reverse since we added from end to start
157     Collections.reverse(path);
158 
159     // reset decorations at end.
160     for (GraphVertex vert : currVertices1.values()) {
161       vert.reset();
162     }
163 
164     return path;
165   }

167   /**
168    * Method for when we want to use conn to update as we go.
169    *
170    * @param start
171    *          starting vertex
172    * @param end
173    *          target vertex
174    * @return List with vertex path
175    */
176   private List<GraphVertex> updateDijkstra(GraphVertex start, GraphVertex end) {
177     // reset caches to load new graph set.
178     this.clear();
179 
180     // dijkstra's algorithm needing to update graph
181     PriorityQueue<GraphVertex> pq = new PriorityQueue<GraphVertex>();
182     start.setDistance(0);
183     pq.add(start);
184     GraphVertex currVertex;
185     GraphVertex nextVertex;
186     boolean found = false;
187 
188     // go through priority queue.
189     while ((currVertex = pq.poll()) != null) {
190       // make sure it hasn't been visited
191       if (currVertex.getVisited()) {
192         continue;
193       }
194 
195       // set as visited
196       currVertex.setVisited(true);
197 
198       // search for it in cache
199       if (currVertices1.keySet().contains(currVertex.getId())) {
200         currVertex = currVertices1.get(currVertex.getId());
201       } else {
202         // if not found, just update from query
203         currVertex = this.updateVertex(currVertex);
204         if (currVertex == null) {
205           // reset decorations at end.
206           for (GraphVertex vert : currVertices1.values()) {
207             vert.reset();
208           }
209           return null;
210         }
211         // add to currVertices1
212         currVertices1.put(currVertex.getId(), currVertex);
213       }
214 
215       // if current search vertex is the end
216       if (currVertex.getId().equals(end.getId())) {
217         found = true;
218         end = currVertex;
219 
220         break;
221       }
222 
223       // for each vertex, loop through edges
224       for (GraphEdge e : currVertex.getEdges()) {
225         nextVertex = e.getTo();
226 
227         if (nextVertex.getVisited()) {
228           continue;
229         }
230 
231         // determine if a better path was found for a vertex.
232         if ((currVertex.getDistance() + e.getWeight()) < nextVertex
233             .getDistance()) {
234           // reset this vertex
235           pq.remove(nextVertex);
236           nextVertex.setDistance((currVertex.getDistance() + e.getWeight()));
237           nextVertex.setPrev(currVertex);
238           pq.add(nextVertex);
239         }
240       }
241     }
242 
243     // now, loop through from end and keep track of which vertex comes before.
244     List<GraphVertex> path = new ArrayList<GraphVertex>();
245     GraphVertex v = end;
246 
247     // if end not found, return empty
248     if (!found) {
249       // reset decorations at end.
250       for (GraphVertex vert : currVertices1.values()) {
251         vert.reset();
252       }
253 
254       return path;
255     }
256     while (v != null) {
257       // update path
258       path.add(v);
259       v = v.getPrev();
260     }
261 
262     // reverse since we added from end to start
263     Collections.reverse(path);
264 
265     // reset decorations at end.
266     for (GraphVertex vert : currVertices1.values()) {
267       vert.reset();
268     }
269 
270     return path;
271   }

273   /**
274    * updateVertex for querying info from database about a vertex.
275    *
276    * @param vert
277    *          vertex to query
278    * @return updated vertex
279    */
280   private GraphVertex updateVertex(GraphVertex vert) {
281     String iden = vert.getId();
282 
283     // set query
284     String query = "SELECT * FROM actor, actor_film, film WHERE actor.id = "
285         .concat(" actor_film.actor AND film.id = actor_film.film").concat(
286             " AND actor.id = ?;");
287 
288     // Create a PreparedStatement
289     PreparedStatement prep = null;
290     ResultSet rs = null;
291     String currName = null;
292     try {
293       prep = conn1.prepareStatement(query);
294 
295       prep.setString(1, iden);
296 
297       // Execute the query and retrieve a ResultStatement
298       rs = prep.executeQuery();
299 
300       // Add the results to a list
301       List<String> actorsMovies = new ArrayList<String>();
302       while (rs.next()) {
303         currName = rs.getString(2);
304         actorsMovies.add(rs.getString(4));
305       }
306       rs.close();
307       prep.close();
308       // make sure this is in db
309       if (actorsMovies.size() == 0) {
310         return vert;
311       }
312 
313       double count = 0.0;
314       // loop through movies found and check for matches
315       query = "SELECT * FROM actor, actor_film, film WHERE actor.id = ".concat(
316           " actor_film.actor AND film.id = actor_film.film").concat(
317               " AND film.id = ?;");
318 
319       // for all matches, try to create vertex/edge as necessary
320       for (String s : actorsMovies) {
321         count = 0;
322         prep = conn1.prepareStatement(query);
323         prep.setString(1, s);
324         rs = prep.executeQuery();
325 
326         List<String[]> results = new ArrayList<String[]>();
327         while (rs.next()) {
328           count += 1.0;
329 
330           // check whether it is itself
331           if (iden.equals(rs.getString(1))) {
332             continue;
333           }
334           // if you also don't have a valid pair
335           if (!this.checkValid(currName, rs.getString(2))) {
336             continue;
337           }
338           // movieData
339           String[] movieData = new String[6];
340 
341           // set initial
342           movieData[0] = s;
343           movieData[1] = rs.getString(6);
344           movieData[2] = iden;
345           movieData[3] = currName;
346           movieData[4] = rs.getString(1);
347           movieData[5] = rs.getString(2);
348           results.add(movieData);
349 
350         }
351 
352         if (Double.compare(count, 0.0) == 0) {
353           // should not be 0 instances of movie, but in case, just continue
354           continue;
355         }
356         // count is weight
357         count = 1.0 / count;
358 
359         for (String[] data : results) {
360           // if all valid, make new instances
361           GraphVertex newVertex;
362 
363           if (currVertices1.keySet().contains(data[4])) {
364             newVertex = currVertices1.get(data[4]);
365           } else {
366             newVertex = new Actor(data[4], data[5]);
367           }
368           GraphEdge newEdge = new Movie(count, vert, newVertex, s, data[1]);
369           vert.addEdges(newEdge);
370 
371           currEdges1.add(data);
372         }
373 
374         prep.close();
375         rs.close();
376       }
377 
378     } catch (SQLException e) {
379       System.err.println("ERROR: Querying Vertices error");
380       vert.reset();
381       vert.setEdges(new HashSet<GraphEdge>());
382       return null;
383     } finally {
384       try {
385         // Close the ResultSet and the PreparedStatement
386         if (prep != null) {
387           prep.close();
388         }
389         if (rs != null) {
390           rs.close();
391         }
392       } catch (SQLException e) {
393         System.err.println("ERROR: Querying Vertices error");
394         vert.reset();
395         vert.setEdges(new HashSet<GraphEdge>());
396         return null;
397       }
398     }
399     return vert;
400   }

402   /**
403    * checks if valid name pair for connection.
404    *
405    * @param currName
406    *          donor
407    * @param s2
408    *          recipient
409    * @return boolean of whether valid or not.
410    */
411   private boolean checkValid(String currName, String s2) {
412     if (currName == null || s2 == null || currName.trim().equals("")
413         || s2.trim().equals("")) {
414       return false;
415     }
416     // check letters of initials
417     String[] temp = currName.replaceAll("\\s+", " ").trim().split(" ");
418     char c1 = temp[temp.length - 1].charAt(0);
419     char c2 = s2.replaceAll("\\s+", " ").trim().split(" ")[0].charAt(0);
420 
421     if (c1 == c2) {
422       return true;
423     }
424     return false;
425   }
