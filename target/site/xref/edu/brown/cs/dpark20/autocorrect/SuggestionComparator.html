1   package edu.brown.cs.dpark20.autocorrect;
2   
3   import java.util.Comparator;
4   
5   /**
6    * SuggestionComparator class, comparator for suggestions to given word/context.
7    *
8    */
9   public class SuggestionComparator implements Comparator<Suggestion> {
10  
11    private String word1;
12    private String prev1;
13  
14    /**
15     * Constructor to set up comparator.
16     *
17     * @param givenWord
18     *          word given to compare to.
19     * @param prevWord
20     *          previous word for bigrams.
21     */
22    public SuggestionComparator(String givenWord, String prevWord) {
23      word1 = givenWord;
24      prev1 = prevWord;
25    }
26  
27    /**
28     * method compare, taking 2 suggestions and finding better one.
29     *
30     * @param s1
31     *          Suggestion one.
32     * @param s2
33     *          Suggestion two.
34     * @return int of result.
35     */
36    @Override
37    public int compare(Suggestion s1, Suggestion s2) {
38      // check to see if either are equal to word1
39      if (s1.getWord().equals(word1)) {
40        return -1;
41      } else if (s2.getWord().equals(word1)) {
42        return 1;
43      }
44  
45      // compare bigram
46      if (prev1 != null) {
47        if (s1.getBigram() > s2.getBigram()) {
48          return -1;
49        } else if (s2.getBigram() > s1.getBigram()) {
50          return 1;
51        }
52      }
53  
54      // compare unigram
55      if (s1.getUnigram() > s2.getUnigram()) {
56        return -1;
57      } else if (s2.getUnigram() > s1.getUnigram()) {
58        return 1;
59      }
60  
61      // compare alphabetically
62      if (s1.getWord().compareTo(s2.getWord()) < 0) {
63        return -1;
64      } else if (s1.getWord().compareTo(s2.getWord()) > 0) {
65        return 1;
66      }
67  
68      return 0;
69    }
70  }
