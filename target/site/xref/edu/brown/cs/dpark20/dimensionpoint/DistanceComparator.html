3   import java.util.Comparator;
4   
5   /**
6    *
7    * Distance Comparator implementing a comparator.
8    *
9    */
10  public class DistanceComparator implements Comparator<DimensionPoint> {
11  
12    private DimensionPoint point1;
13  
14    /**
15     * Sets point to compare to.
16     *
17     * @param point
18     *          point to store for comparison
19     */
20    public DistanceComparator(DimensionPoint point) {
21      point1 = point;
22  
23    }
24  
25    /**
26     * Override compare method to compare distances.
27     *
28     * @param o1
29     *          first point
30     * @param o2
31     *          second point
32     * @return int comparison
33     */
34    @Override
35    public int compare(DimensionPoint o1, DimensionPoint o2) {
36  
37      // we should ensure that the o1 and o2 passed in are the same
38      // dimension as each other, as well as point1.
39  
40      double distance1 = o1.distance(point1);
41      double distance2 = o2.distance(point1);
42  
43      if (distance1 > distance2) {
44        return 1;
45      } else if (distance1 < distance2) {
46        return -1;
47      }
48  
49      return 0;
50    }
51  
52  }
