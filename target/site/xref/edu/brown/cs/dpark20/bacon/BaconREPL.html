48    /**
49     * Method to evaluate the command given.
50     *
51     * @param cmd
52     *          command with spaces
53     * @param shouldntPrint
54     *          whether need to print out results or not
55     * @param gui
56     *          whether running gui or not
57     * @return int to say if ran anything or not
58     */
59    public int parseAndEvaluate(String cmd, boolean shouldntPrint, boolean gui) {
60      assert (cmd != null);
61  
62      String noSpace = cmd.replaceAll("\\s+", " ").trim();
63      String[] commandWords = noSpace.split(" ");
64      int len = commandWords.length;
65  
66      // LOOP THROUGH ALL POSSIBILE COMMANDS, EVALUATING
67  
68      if (commandWords[0].equals("mdb")) {
69        // MDB COMMAND
70  
71        // if len is not 2, then it is the wrong command
72        if (len != 2) {
73          System.err.println("ERROR: Incorrect number of arguments; "
74              .concat("expected <mdb sql_database_file>"));
75          return 1;
76        }
77        // process the command, using command without "mdb " as filename
78        this.mdb(noSpace.substring(commandWords[0].length() + 1).trim(), gui);
79  
80      } else if (commandWords[0].equals("connect")) {
81        // CONNECT COMMAND
82  
83        // if len is less than 3, then it is the wrong command
84        if (len < 3) {
85          System.err.println("ERROR: Incorrect number of arguments; "
86              .concat("expected <connect \"name1\" \"name2\">"));
87          return 1;
88        }
89        // process the command, first identifying names
90  
91        char c;
92        int count = 0;
93        int[] qLoc = new int[4];
94        // everything after the "command "
95        String names;
96        names = cmd.trim().substring(commandWords[0].length() + 1).trim();
97  
98        // check to make sure command is formatted correctly
99        for (int i = 0; i < names.length(); i++) {
100         c = names.charAt(i);
101         if (c == '"') {
102           if (count >= 4) {
103             System.err.println("ERROR: Incorrect args format; "
104                 .concat("expected <connect \"name1\" \"name2\">"));
105             return 1;
106           }
107           qLoc[count] = i;
108           count++;
109         }
110       }
111 
112       // ensure the quotes were well-formatted and in right spots
113       if (count != 4 || qLoc[0] != 0 || qLoc[3] != (names.length() - 1)
114           || names.substring(qLoc[1] + 1, qLoc[2]).length() == 0
115           || names.substring(qLoc[1] + 1, qLoc[2]).trim().length() != 0) {
116 
117         System.err.println("ERROR: Incorrect args format; "
118             .concat("expected <connect \"name1\" \"name2\">"));
119         return 1;
120       }
121 
122       // formatted fine, so process command by getting names given
123       this.connect(names.substring(qLoc[0] + 1, qLoc[1]),
124           names.substring(qLoc[2] + 1, qLoc[3]), shouldntPrint);
125 
126     } else {
127       // not bacon command
128       return 0;
129     }
130 
131     return 1;
132   }

134   /**
135    * mdb method for loading database.
136    *
137    * @param file
138    *          file to load in sql
139    */
140   private void mdb(String file, boolean gui) {
141     assert (file != null);
142 
143     // check to see if it is a file
144     File f = new File(file);
145     if (!f.isFile() || f.isDirectory()
146         || !Files.getFileExtension(file).equals("sqlite3")) {
147       System.err.println("ERROR: File given is not valid");
148       return;
149     }
150 
151     // Set up a connection and store it in a field
152     try {
153       Class.forName("org.sqlite.JDBC");
154     } catch (ClassNotFoundException e) {
155       System.err.println("ERROR: Class not found");
156       return;
157     }
158     String url = "jdbc:sqlite:" + file;
159     try {
160       // close current connection if not null
161       if (conn1 != null) {
162         conn1.close();
163       }
164       conn1 = DriverManager.getConnection(url);
165       Statement stat = conn1.createStatement();
166       stat.executeUpdate("PRAGMA foreign_keys = ON;");
167       stat.close();
168     } catch (SQLException e) {
169       // catch possible error
170       if (conn1 != null) {
171         try {
172           conn1.close();
173         } catch (SQLException e1) {
174           System.err.println("ERROR: SQL setup error in mdb");
175         }
176       }
177       conn1 = null;
178       System.err.println("ERROR: SQL setup error in mdb");
179       return;
180     }
181 
182     System.out.println(String.format("db set to %s", file));
183 
184     // Now, loaded file, so add to query
185     if (query1 != null) {
186       // close first if needed;
187       query1.clear();
188     }
189     query1 = new Dijkstras(conn1);
190 
191     if (gui) {
192       this.setUpAutocorrect("data/bacon/names.txt");
193     }
194   }

196   /**
197    * connect method to take in 2 names and find connection.
198    *
199    * @param name1
200    *          first name
201    * @param name2
202    *          second name
203    */
204   private void connect(String name1, String name2, boolean notPrint) {
205     assert (name1 != null);
206     assert (name2 != null);
207 
208     // check to make sure db was loaded
209     if (conn1 == null || query1 == null) {
210       System.err.println("ERROR: No query database loaded yet");
211       return;
212     }
213 
214     // results must be reset first
215     results1.clear();
216 
217     // easily check if it's the same name
218     if (name1.equals(name2)) {
219       if (!notPrint) {
220         System.out.println(String.format("%s is themself!", name1));
221       }
222       return;
223     }
224 
225     Actor start = null;
226     Actor end = null;
227 
228     // first check if the search items' IDs are cached from before
229     for (String[] e : query1.getEdges()) {
230       // based on edges cache, formatted with string info stored in array
231       if (e.length != 6) {
232         continue;
233       }
234       if (e[3].equals(name1)) {
235         start = new Actor(e[2], name1);
236       }
237       if (e[3].equals(name2)) {
238         end = new Actor(e[2], name2);
239       }
240       if (e[5].equals(name1)) {
241         start = new Actor(e[4], name1);
242       }
243       if (e[5].equals(name2)) {
244         end = new Actor(e[4], name2);
245       }
246     }
247 
248     // set query
249     String query = "SELECT * FROM actor WHERE actor.name = ?;";
250 
251     // Create a PreparedStatement
252     PreparedStatement prep = null;
253     ResultSet rs = null;
254 
255     // query db for it now
256     try {
257 
258       // only need to query if start/end are null
259 
260       if (start == null) {
261         prep = conn1.prepareStatement(query);
262 
263         prep.setString(1, name1);
264 
265         // Execute the query and retrieve a ResultStatement
266         rs = prep.executeQuery();
267 
268         while (rs.next()) {
269           // just take the first one found, if any.
270           start = new Actor(rs.getString(1), rs.getString(2));
271           break;
272         }
273         rs.close();
274         prep.close();
275       }
276 
277       if (end == null) {
278         // repeat for end vertex
279         prep = conn1.prepareStatement(query);
280 
281         prep.setString(1, name2);
282 
283         // Execute the query and retrieve a ResultStatement
284         rs = prep.executeQuery();
285 
286         while (rs.next()) {
287           // just take the first one found, if any.
288           end = new Actor(rs.getString(1), rs.getString(2));
289           break;
290         }
291         rs.close();
292         prep.close();
293 
294       }
295 
296       // check if either start or end are null
297       if (start == null) {
298         System.err.println("ERROR: Search for start actor unsuccessful");
299         return;
300       } else if (end == null) {
301         System.err.println("ERROR: Search for end actor unsuccessful");
302         return;
303       }
304 
305       // get path found in dijkstra's
306       List<GraphVertex> path = query1.graphQuery(start, end);
307 
308       // Print out what's needed
309       this.displayResult(name1, name2, start, end, path, notPrint);
310 
311     } catch (SQLException e) {
312       System.err.println("ERROR: Querying Connect error");
313     } finally {
314       try {
315         // Close the ResultSet and the PreparedStatement
316         if (prep != null) {
317           prep.close();
318         }
319         if (rs != null) {
320           rs.close();
321         }
322       } catch (SQLException e) {
323         System.err.println("ERROR: Querying Connect error");
324       }
325     }
326 
327   }

329   private void displayResult(String name1, String name2, Actor start, Actor end,
330       List<GraphVertex> path, boolean notPrint) {
331     // check edge cases
332     if (path == null) {
333       return;
334     }
335 
336     // shouldn't get size 1 path
337     if (path.size() == 1) {
338       System.err.println("ERROR: Path is too small");
339       return;
340     } else if (path.size() == 0) {
341       // no path found
342       if (!notPrint) {
343         System.out.println(String.format("%s -/- %s", name1, name2));
344       } else {
345         StringBuilder s = new StringBuilder();
346         s.append(name1);
347         s.append(" -/- ");
348         s.append(name2);
349         results1.addVertices(start);
350         results1.addVertices(end);
351       }
352     }
353 
354     // Create List of Strings to find output values
355     List<String> outputVals = null;
356     results1.addVertices(start);
357     for (int i = 0; i < path.size() - 1; i++) {
358       // get values to output
359 
360       outputVals = this.getInfo(path.get(i), path.get(i + 1));
361 
362       // should be size 3 for 3 pieces of info
363       if (outputVals == null || outputVals.size() != 4) {
364         return;
365       }
366 
367       assert (outputVals.get(0) != null);
368       assert (outputVals.get(1) != null);
369       assert (outputVals.get(2) != null);
370       assert (outputVals.get(3) != null);
371 
372       // build output
373       StringBuilder build = new StringBuilder();
374       build.append(outputVals.get(0));
375       build.append(" -> ");
376       build.append(outputVals.get(1));
377       build.append(" : ");
378       build.append(outputVals.get(2));
379       // append concatenated result output to results list
380       results1.addVertices((Actor) path.get(i + 1));
381       results1.addEdges(new Movie(0.0, path.get(i), path.get(i + 1),
382           outputVals.get(3), outputVals.get(2)));
383 
384       // print out if not being displayed in gui
385       if (!notPrint) {
386         System.out.println(build);
387       }
388     }
389 
390   }

392   /**
393    * Method to get info to print from vertices.
394    *
395    * @param v1
396    *          first vertex
397    * @param v2
398    *          second vertex
399    * @return List with output strings.
400    */
401   private List<String> getInfo(GraphVertex v1, GraphVertex v2) {
402     String mov = null;
403     double min = Double.MAX_VALUE;
404     String givenId = v2.getId();
405     for (GraphEdge e : v1.getEdges()) {
406       if (e.getTo().getId().equals(givenId) && e.getWeight() < min) {
407         min = e.getWeight();
408         mov = e.getId();
409       }
410     }
411 
412     // args to return
413     List<String> args = new ArrayList<>();
414 
415     // search through cache first
416     for (String[] movie : query1.getEdges()) {
417       // if somehow wrong format, continue
418       if (movie.length != 6) {
419         continue;
420       }
421       // if found movie match
422       if (mov.equals(movie[0]) && v1.getId().equals(movie[2])
423           && v2.getId().equals(movie[4])) {
424         // found, so add args in format name1, name2, movie, movieId
425         args.add(movie[3]);
426         args.add(movie[5]);
427         args.add(movie[1]);
428         args.add(movie[0]);
429         return args;
430       }
431     }
432 
433     return args;
434   }

461

465         }
466         writer.close();
467       } catch (IOException e) {
468         System.err.println("ERROR: Reading and making file");
469       }
470 
471     } catch (SQLException e) {
472       System.err.println("ERROR: Querying Actors error");

474       try {
475         // Close the ResultSet and the PreparedStatement
476         if (prep != null) {
477           prep.close();
478         }
479         if (rs != null) {
480           rs.close();
481         }
482       } catch (SQLException e) {
483         System.err.println("ERROR: Querying Actors SQL Closing error");
484       }
485     }
486     // set up acrepl, with defaults prefix, whitespace on and led 1
487     acrepl1.parseAndEvaluate(String.format("corpus %s", file),
488         String.format("corpus %s", file), true);
